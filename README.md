1. Generating Arrays and Samples

Use ExperimentConfigurator to create all datasets, including:
    / full arrays generated by the selected models
    / sample sets produced via cluster, stratified, and combined sampling strategies

    int main() {
        const int n = 10000;
        const int minValue = 1;
        const int maxValue = 10000;

        const std::string root =
            "C:/Users/markg/CLionProjects/DisorderMetrics/src/experiment_data_input";

        const int countPerSet = 1000;

        std::vector<int> kValues = {2500, 5000, 7500, 9000};
        std::vector<int> uValues = {2500, 5000, 7500, 9000};


        const int sqrt_n   = static_cast<int>(std::floor(std::sqrt((double)n)));
        const int log2_n   = static_cast<int>(std::floor(std::log2((double)n)));
        const int n_div_25 = static_cast<int>(std::lround(n / 2.5));

        std::vector<int> runsValues = {
            13, 100, 200, 1000, 2500,
            sqrt_n,
            2 * sqrt_n,
            log2_n,
            n / 4,
            n / 10,
            n / 20,
            n_div_25
        };

        try {
            const int S = 100;
            ExperimentConfigurator conf(n, S, minValue, maxValue, root);

            conf.configure(
                countPerSet,
                kValues,
                uValues,
                runsValues,
                /*includePermutation*/         true,
                /*includeMultisetPermutation*/ true,
                /*includeRandomK*/             true,
                /*includeRuns*/                true,
                /*overwrite*/                  true
            );

            std::cout << "Done: sampleSize = " << S << "\n";
        }
        catch (const std::exception& e) {
            std::cerr << "Error: " << e.what() << "\n";
        }

2. Compute Presortedness Metrics

Use Evaluator to compute normalized metrics for:
    full arrays (arrays.csv)
    all sampled arrays (samples.csv)

    int main() {

        std::string input  = "src/experiment_data_input";
        std::string output = "src/experiment_data_output";

        Evaluator ev;
        ev.prepareOutputStructure(input, output);
        ev.evaluateAll(input, output, true); // overwrite results
    }

Output is placed in a mirrored directory structure:
